def Re30_1800(Re):
    '''
    compute Nusselt Number for Reynolds Number between 30 and 1800
    '''
    up = Re
    down = (1.08 * np.power(Re,1.22)) - 5.2
    return up/down

def Re1800(Re,Pr):
    '''
    compute Nusselt Number for Reynolds Number more than 1800
    '''
    up = Re
    down = 8750 + (58 * np.power(Pr,-0.5) * (np.power(Re, 0.75) - 253))
    return up/down

def RE(Re, Pr):
    if Re <= 30:
        return Re30(Re)
    elif 30 < Re <= 1800:
        return Re30_1800(Re)
    else:
        return Re1800(Re,Pr)

def res(array, Pr):
    b = []
    for i in array:
        b.append(RE(i,Pr))
    return np.array(b)
